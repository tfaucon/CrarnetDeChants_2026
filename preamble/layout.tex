% --- Numéro de page décoré par le foulard de Tassin ---
% Le foulard est placé en overlay (hors flux) pour ne réserver aucun espace
% vertical dans l'en-tête. Le numéro est superposé sur le nœud (woggle).
% anchor = coin de la page où accrocher l'image (north east / north west).
\newcommand{\foulardpagenumLeft}{%
  \begin{tikzpicture}[remember picture, overlay]
    \node[anchor=north west, inner sep=0pt, xshift=6mm, yshift=-6mm]
      (foulard) at (current page.north west)
      {\includegraphics[height=14mm]{images/foulardTassin}};
    \node[anchor=center, font=\small\bfseries, color=black]
      at ([yshift=3.5mm]foulard.center) {\thepage};
  \end{tikzpicture}%
}
\newcommand{\foulardpagenumRight}{%
  \begin{tikzpicture}[remember picture, overlay]
    \node[anchor=north east, inner sep=0pt, xshift=-6mm, yshift=-6mm]
      (foulard) at (current page.north east)
      {\includegraphics[height=14mm]{images/foulardTassin}};
    \node[anchor=center, font=\small\bfseries, color=black]
      at ([yshift=3.5mm]foulard.center) {\thepage};
  \end{tikzpicture}%
}

% --- En-têtes / pieds de page modernes ---
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{\foulardpagenumLeft}
\fancyhead[RO]{\foulardpagenumRight}
\fancyhead[RE,LO]{\small\color{tassindarkbrown}\textit{\leftmark}}
\renewcommand{\headrulewidth}{0pt}
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyhead[LE]{\foulardpagenumLeft}%
  \fancyhead[RO]{\foulardpagenumRight}%
  \renewcommand{\headrulewidth}{0pt}%
}
