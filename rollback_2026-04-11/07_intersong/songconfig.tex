% Pas de numéros de chants, mais numéros de couplets
\nosongnumbers

% Autoriser les chants à démarrer n'importe où (pas forcément en haut de colonne).
% Réduit drastiquement les blancs de queue.
\songpos{0}

% Réduire l'espace entre numéro de couplet et texte
\renewcommand{\printversenum}[1]{\lyricfont#1.\,}
\settowidth{\versenumwidth}{\printversenum{9\kern.4em}}

% Espacement entre colonnes
\setlength{\columnsep}{12pt}

% Espacement entre chants et couplets (resserré)
\versesep=3pt plus 2pt minus 1pt
\afterpreludeskip=6pt plus 4pt minus 2pt
\beforepostludeskip=6pt plus 4pt minus 2pt

% Marge d'étirement pour absorber les lignes borderline en 3 colonnes
\setlength{\emergencystretch}{3em}

% Espacement vertical (accords au-dessus des paroles)
\baselineadj=-2pt

% Barre de refrain en vert Tassin
% Use the built-in chorus bar (no extra rule from \chorusmark)
\setlength{\cbarwidth}{1.2pt}
\setlength{\sbarheight}{0pt}
\renewcommand{\chorusmark}{}

% Patch the built-in chorus bar to be green
\makeatletter
\renewcommand\SB@chorusbar[1]{%
  \ifdim\cbarwidth>\z@%
    \SB@dimen\ht#1%
    \SB@dimenii\dp#1%
    \advance\SB@dimen%
      \ifSB@chorustop\ifchorded\else2\fi\fi\SB@dimenii%
    \SB@skip\SB@dimen\relax%
    \SB@computess\SB@skip1\@plus#1%
    \SB@computess\SB@skip{-1}\@minus#1%
    \nointerlineskip\null\nobreak%
    {\color{tassingreen}\leaders\vrule\@width\cbarwidth\vskip\SB@skip}%
    \ifSB@chorustop\ifchorded\else%
      \advance\SB@skip-\SB@dimenii%
    \fi\fi%
    \nobreak\vskip-\SB@skip%
  \fi%
}
\makeatother

% Chorus font: bold, no color change (prevent green bleed into following verses)
\renewcommand{\chorusfont}{\sffamily\footnotesize\bfseries}
