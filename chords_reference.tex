% =============================================================================
%  CHORD DIAGRAM DRAWING HELPERS
%  Coordinate system: x = 0..5 (string 6=low E at x=0, string 1=high e at x=5)
%                     y = 0 (nut/top) down to -4 (4th fret)
%  Dots placed at (x, -fret + 0.5) so they sit in the middle of the fret space
% =============================================================================

% Draw a dot: \dot{string 6..1}{fret 1..4}
\newcommand{\chmark}[2]{%
  \pgfmathtruncatemacro{\xp}{6-#1}%
  \node[circle, fill=tassindarkbrown, minimum size=3pt, inner sep=0pt] at (\xp, {-#2+0.5}) {};%
}

% Draw a barre: \barre{from string}{to string}{fret}
\newcommand{\barre}[3]{%
  \pgfmathtruncatemacro{\xa}{6-#1}%
  \pgfmathtruncatemacro{\xb}{6-#2}%
  \draw[tassindarkbrown, line width=2pt, line cap=round] (\xb, {-#3+0.5}) -- (\xa, {-#3+0.5});%
}

% Open string marker: \omark{string 6..1}
\newcommand{\omark}[1]{%
  \pgfmathtruncatemacro{\xp}{6-#1}%
  \node[draw=tassingreen, circle, inner sep=0.8pt, line width=0.35pt, font=\tiny]
    at (\xp, 0.45) {};%
}

% Muted string marker: \xmark{string 6..1}
\newcommand{\xmark}[1]{%
  \pgfmathtruncatemacro{\xp}{6-#1}%
  \node[font=\tiny\bfseries, text=tassinred] at (\xp, 0.45) {\textsf{x}};%
}

% Fret number indicator: \fretnum{number}
\newcommand{\fretnum}[1]{%
  \node[anchor=east, font=\sffamily\tiny\color{tassindarkbrown}] at (-0.3, -0.5) {#1\,fr};%
}

% Spacing between chord diagrams — fixed-width box for regularity
\newlength{\chordwidth}
\setlength{\chordwidth}{25mm}
\newcommand{\cspace}{\hspace{0pt}}

% Section header for root note groups
\newcommand{\rootsection}[1]{%
  \vspace{3pt}%
  {\sffamily\large\bfseries\color{tassingreen}#1}\,%
  \textcolor{tassinlightbrown}{\leaders\hrule height 0.5pt\hfill\kern0pt}%
  \vspace{2pt}\par\noindent%
}

% =============================================================================
%  CHORD DIAGRAM ENVIRONMENT
%  Usage: \begin{chordbox}{Name}{startfret} ... \end{chordbox}
%  startfret=0 means nut shown; >0 means fret number shown
%  Inside: use \chmark, \barre, \omark, \xmark
% =============================================================================
\newenvironment{chordbox}[2]{%
  \begin{minipage}[t]{\chordwidth}\centering\begin{tikzpicture}[scale=0.42, every node/.style={inner sep=0pt}]
    % Draw grid: 4 frets, 6 strings
    \foreach \f in {0,...,4} {
      \draw[tassinbrowngray, line width=0.3pt] (0,{-\f}) -- (5,{-\f});
    }
    \foreach \s in {0,...,5} {
      \draw[tassinbrowngray, line width=0.3pt] (\s,0) -- (\s,-4);
    }
    % Nut or fret number
    \ifnum#2=0
      \draw[tassindarkbrown, line width=1.5pt] (-0.05,0.03) -- (5.05,0.03);
    \else
      \fretnum{#2}
    \fi
    % Chord name below
    \node[anchor=north, font=\sffamily\bfseries\footnotesize, text=tassinred]
      at (2.5, -4.45) {#1};
}{%
  \end{tikzpicture}\end{minipage}%
}
